
CREATE DATABASE IF NOT EXISTS sistema_ventas;
USE sistema_ventas;


CREATE TABLE Proveedores (
    id INT AUTO_INCREMENT PRIMARY KEY,
    RUT VARCHAR(20) UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    direccion VARCHAR(255) NOT NULL,
    telefono VARCHAR(20),
    pagina_web VARCHAR(100)
);


CREATE TABLE Clientes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    RUT VARCHAR(20) UNIQUE,
    nombre VARCHAR(100) NOT NULL,
    direccion_calle VARCHAR(100) NOT NULL,
    direccion_numero INT NOT NULL,
    direccion_comuna VARCHAR(50) NOT NULL,
    direccion_ciudad VARCHAR(50) NOT NULL
);


CREATE TABLE TelefonosClientes (
    id INT AUTO_INCREMENT PRIMARY KEY,
    RUT_cliente VARCHAR(20),
    telefono VARCHAR(20) NULL,
    FOREIGN KEY (RUT_cliente) REFERENCES Clientes(RUT)
);


CREATE TABLE Categorias (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(50) NOT NULL,
    descripcion VARCHAR(255) NULL
);


CREATE TABLE Productos (
    id INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    precio_actual DECIMAL(10, 2) NOT NULL,
    stock INT NOT NULL,
    proveedor_id INT NULL,
    categoria_id INT,
    FOREIGN KEY (proveedor_id) REFERENCES Proveedores(id),
    FOREIGN KEY (categoria_id) REFERENCES Categorias(id)
);


CREATE TABLE Ventas (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fecha DATE NOT NULL,
    RUT_cliente VARCHAR(20) NULL,
    descuento DECIMAL(5, 2) NULL,
    monto_final DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (RUT_cliente) REFERENCES Clientes(RUT)
);


CREATE TABLE DetalleVenta (
    id INT AUTO_INCREMENT PRIMARY KEY,
    venta_id INT,
    producto_id INT,
    precio_venta DECIMAL(10, 2) NOT NULL,
    cantidad INT NOT NULL,
    monto_total DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (venta_id) REFERENCES Ventas(id),
    FOREIGN KEY (producto_id) REFERENCES Productos(id)
);
